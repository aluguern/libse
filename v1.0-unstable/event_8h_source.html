00040 class Event {
00041 private:
00042   // On 32-bit architectures, the legal range of identifiers is 0 to 2^15-1.
00043   static size_t s_next_id;
00044 
00045   const size_t m_event_id;
00046   const unsigned m_thread_id;
00047   const bool m_is_read;
00048   const Type* const m_type_ptr;
00049   const MemoryAddr m_addr;
00050   const std::shared_ptr<ReadInstr<bool>> m_condition_ptr;
00051 
00052   // Even positive integers (m = 2k for k in 0 to 2^15-1) are for read
00053   // events. In turn, odd positive integers (n = 2k + 1 for k in 0 to 2^15-1)
00054   // are write events. That is, the maximal read event identifier is 2^30-2
00055   // and the maximal write event identifier is 2^30-1. These upper limits
00056   // stem from Z3 which aligns char pointers for symbol names by 4 bytes on
00057   // 32-bit architectures.
00058   static size_t next_id(bool is_read) {
00059     const size_t even = 2 * s_next_id++;
00060     return is_read ? even : even + 1;
00061   }
00062 
00063 protected:
00064
